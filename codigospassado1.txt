// Definindo Classe

class Personagem{

    // PROPRIEDADES: São as características (o que ele TEM)
    public string Nome;
    public int Vida;

    // MÉTODOS: São os comportamentos (o que ele FAZ)
    public void Apresentar(){
        Console.WriteLine($"Olá, meu nome é {Nome} e tenho {Vida} de HP.");
    }

    public void Atacar(Personagem alvo){
        Console.WriteLine($"{Nome} atacou {alvo.Nome}!");
        alvo.Vida -= 10;
        Console.WriteLine($"Agora {alvo.Nome} tem {alvo.Vida} de HP!");
    }
}


// Conta Bancaria
ContaBancaria minhaConta = new ContaBancaria("Victor", 500);
Console.WriteLine($"Conta de {minhaConta.Titular} criado com sucesso!");

ContaBancaria outraConta = new ContaBancaria("Amigo", 100);
Console.WriteLine($"Conta de {outraConta.Titular} criado com sucesso!");

Console.WriteLine($"Saldo inicial de {minhaConta.Titular}: {minhaConta.Saldo}");
Console.WriteLine($"Saldo inicial de {outraConta.Titular}: {outraConta.Saldo}");

minhaConta.Transferir(200, outraConta);

Console.WriteLine($"Saldo final de {minhaConta.Titular}: {minhaConta.Saldo}");
Console.WriteLine($"Saldo final de {outraConta.Titular}: {outraConta.Saldo}");

class ContaBancaria(string NomeInformado, double SaldoInicial)
{
    public string Titular = NomeInformado;
    public double Saldo = SaldoInicial;
    public void Depositar(double Valor)
    {
        Saldo += Valor;
        Console.WriteLine(Saldo);
    }

    public void Transferir(double Valor, ContaBancaria ContaDestino)
    {
        if(this.Saldo >= Valor)
        {
            this.Saldo -= Valor;
            ContaDestino.Depositar(Valor);
        }
        else{
            Console.WriteLine("Saldo insuficiente.");
        }
    }
}

public abstract class Veiculo{

    protected string Placa{get; set;}
    public string Modelo{get; set;}

    public abstract void Abastcer(double litros);
}

public interface IManutencao
{
    void RealizarRevisao();
}

public class Caminhao : Veiculo, IManutencao{
    public double CargaMaxima{get; set;}

    public override void Abastecer(double litros){
        Console.WriteLine($"Abastecendo Caminhão com {litros} litros de Diesel. Custo: R${litros * 6}.")
    }

    public void RealizarRevisao(){
        Console.WriteLine("Revisando freios e suspensão do Caminhão")
    }

}

public class CarroEletrico : Veiculo, IManutencao{
    public override void Abastecer(double litros){
        Console.WriteLine($"Carregando bateria com {litros}kWh");
    }

    public void RealizarRevisao(){
        Console.WriteLine("Verificando celulas da bateria e software.");
    }
}
